## Features

- Arrow navigation should select caption within a block.

</PackageInfo>

## Installation

```bash
npm install @udecode/plate-caption
```

## Usage

```tsx
import { createCaptionPlugin } from '@udecode/plate-caption';

const plugins = [
  // ...otherPlugins,
  createCaptionPlugin({
    options: { pluginKeys: [ELEMENT_IMAGE, ELEMENT_MEDIA_EMBED] },
  }),
];
```

## API

### createCaptionPlugin

<APIOptions>
  <APIItem name="pluginKeys" type="string[]" required>
    The plugin key of the blocks that have caption.
  </APIItem>
</APIOptions>

### TCaptionElement

<APIAttributes>
Extends `TElement`.

<APIItem name="caption" type="TDescendant[]" optional>
Caption value.
</APIItem>
</APIAttributes>
